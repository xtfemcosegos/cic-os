/**
 * Extrae de forma segura la lista de todos los nombres de casetas
 * desde la base de datos IndexedDB 'cic-os' -> almacen 'casetas'.
 * 
 * @returns {Promise<string[]>} Promesa que resuelve con el array de nombres de casetas.
 */
function extraerNombresDeCasetas() {
    return new Promise((resolve, reject) => {
        // Abrimos la base de datos sin forzar versión para usar la activa
        const solicitud = indexedDB.open("cic-os");

        solicitud.onerror = (evento) => {
            reject("Error al conectar con IndexedDB: " + evento.target.error.message);
        };

        solicitud.onsuccess = (evento) => {
            const db = evento.target.result;

            // Validamos existencia del almacén solicitado
            if (!db.objectStoreNames.contains("casetas")) {
                db.close();
                reject("El almacén 'casetas' no existe en la base de datos.");
                return;
            }

            const transaccion = db.transaction("casetas", "readonly");
            const almacen = transaccion.objectStore("casetas");
            
            // Dado que las casetas usan el 'nombre' como clave única, extraemos todas las llaves primarias
            const solicitudLlaves = almacen.getAllKeys();

            solicitudLlaves.onsuccess = (e) => {
                const nombres = e.target.result;
                db.close();
                resolve(nombres); // Retorna array de strings (ej: ["Caseta Norte", "Caseta Sur"])
            };

            solicitudLlaves.onerror = (e) => {
                db.close();
                reject("Error de lectura en almacén: " + e.target.error.message);
            };
        };
    });
}
